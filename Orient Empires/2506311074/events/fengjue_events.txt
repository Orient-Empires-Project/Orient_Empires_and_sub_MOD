namespace = fengjue
#给予自定义绰号
fengjue.1 = {
	type = character_event
	title = fengjue.1.t
	desc = fengjue.1.desc
	theme = crown
	left_portrait = {
		character = root
	}
	right_portrait = {
		character = root
	}
	immediate = {	
	}

	widget = {
		gui = "event_window_widget_enter_nickname"
		container = "dynamic_birth_name"
		controller = {
			type = text
			data = {
				key = fengjue
				default = {
					localization_key = fengjue_wei
				}
				default = {
					localization_key = fengjue_shu
				}
				default = {
					localization_key = fengjue_wu
				}
			}
		}
		setup_scope = {
			root = { save_scope_as = text_target }
		}
	}
	option = { 
		name = newnianhao.a
	}
	after = {	
		if = {
			limit = { NOT = { has_variable = fengjue_var } } 
			set_variable = {
				name = fengjue_var
				value = flag:custom
			}
		}
		create_dynamic_title = {
			tier = kingdom
			name = fengjue_var_text
		}
		create_title_and_vassal_change = {
			type = created
			save_scope_as = change
			add_claim_on_loss = no
		}		
		scope:new_title = {
			generate_coa = yes 
			change_title_holder = {
				holder = root
				change = scope:change
			}
		}
		resolve_title_and_vassal_change = scope:change
	}
}

fengjue.2 = {
	type = character_event
	title = fengjue.2.t
	desc = fengjue.2.desc
	theme = crown
	left_portrait = {
		character = scope:actor
	}
	right_portrait = {
		character = scope:recipient
	}
	immediate = {
		scope:target = {
			save_scope_as = old_title
		}
	}

	option = {
		name = grant_kingdom_guo

		trigger = {
			scope:kingdom = yes
		}
		create_kingdom_guo_effect = yes
		scope:recipient = {
			add_realm_law_skip_effects = partition_succession_law

			add_opinion_to_actor_receive_kingdom = yes
		}
	}

	option = {
		name = grant_kingdom_jun

		trigger = {
			scope:kingdom = yes
		}

		scope:recipient = {
			create_kingdom_jun_effect = yes

			add_realm_law_skip_effects = partition_succession_law

			add_opinion_to_actor_receive_kingdom = yes
		}

	}

	option = {
		name = grant_duchy_guo
		trigger = {
			NAND = {
				scope:recipient.primary_title ?= { has_variable = titular_title }
				scope:recipient.primary_title ?= { tier = tier_duchy }
			}
			scope:duchy = yes
		}
		create_duchy_guo_effect = yes
		scope:recipient = {
			add_realm_law_skip_effects = partition_succession_law

			add_opinion_to_actor_receive_duchy = yes
		}
	}

	option = {
		name = grant_duchy_jun

		trigger = {
			NAND = {
				scope:recipient.primary_title ?= { has_variable = titular_title }
				scope:recipient.primary_title ?= { tier = tier_duchy }
			}
			scope:duchy = yes
		}
		create_duchy_jun_effect = yes
		scope:recipient = {
			add_realm_law_skip_effects = partition_succession_law

			add_opinion_to_actor_receive_duchy = yes
		}
	}

	#自定义
	option = {
		name = guohao.002.title
		trigger = {
			exists = scope:target
		}
		scope:recipient = {
			save_scope_as = holder
		}
		scope:actor = {
			hidden_effect = {
				if = {
					limit = {
						scope:kingdom = yes
					}
					add_prestige = 500
				}
				else = {
					add_prestige = 200
				}
				add_piety = {
					value = 200
				}
			}
			trigger_event = fengjue.3
		}
	}

	option = {
		name = cancel

		scope:actor = {
			hidden_effect = {
				if = {
					limit = {
						scope:kingdom = yes
					}
					add_prestige = 500
				}
				else = {
					add_prestige = 200
				}
				add_piety = {
					value = 200
				}
			}
		}
	}

	after = {
		scope:recipient = { remove_character_flag = titular_holder }
		
	}

}

	title = fengjue.4.t

	}

		character = scope:holder

		if = {

		else = {
			root = { save_scope_as = holder }
		}
	}

	option = {
		name = found_kingdom_decision
		trigger_event = fengjue.5
		add_prestige = -500
		add_piety = -200
	}
	option = {
		name = found_duchy_decision
		trigger_event = fengjue.6
		add_prestige = -200
		add_piety = -200
	}

	option = { 
		name = guohao.2.b
		is_cancel_option = yes
	}

	after = {
	}

}
