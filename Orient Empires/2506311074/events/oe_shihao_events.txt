namespace = shihao

shihao.1 = {
	type = character_event
	hidden = yes
	trigger = {
		OR = {
		has_title = title:e_celestia
		has_title = title:e_tang
		has_title = title:e_song
		AND = {
			OR = {
				has_culture_group = culture_group:chinese_group
				has_culture = culture:khitan
				has_culture = culture:shatuo
				has_culture = culture:tuyuhun
				has_culture = culture:min
				has_culture = culture:jiaozhi
				has_culture = culture:balhae
				has_culture = culture:jurchen
				has_culture_group = culture_group:goryeo_group
			}
				highest_held_title_tier > tier_kingdom
				has_government = imperial_government
			}	
		}
	}
	immediate = {
		if = {
			limit = {
				OR = {
					has_trait = education_martial_4
					has_trait = education_diplomacy_4
					has_trait = education_stewardship_4
				}
				martial >= 18
				diplomacy >= 18
				stewardship >= 12
			}
			shihao_1st_effect = yes
		}
		if = {
			limit = {
			OR = {
				has_trait = education_martial_3
				has_trait = education_martial_4
				}
				martial >= 20
				stewardship >= 12
			}
			shihao_2nd_effect = yes
		}
		else_if = {
			limit = {
			OR = {
				has_trait = education_diplomacy_3
				has_trait = education_diplomacy_4
				}
				diplomacy >= 20
				stewardship >= 12
			}
			shihao_3th_effect = yes
		}
		else_if = {
			limit = {
				diplomacy >= 15
				stewardship >= 15
			}
			give_nickname = "shihao_ding"
		}
		else_if = {
			limit = {
				has_trait = diligent
				OR = {
					has_trait = patient
					has_trait = generous
				}
				OR = {
					diplomacy >= 12
					stewardship >= 12
				}
			}
			shihao_4th_effect = yes
		}
		else_if = {
			limit = {
				has_trait = generous
				has_trait = humble
				OR = {
					diplomacy >= 12
					stewardship >= 12
				}
			}
			shihao_4th_effect = yes
		}
		else_if = {
			limit = {
			OR = {
			AND = {
				has_trait = education_learning_2
				learning >= 12
				}	
				OR = {
					has_trait = education_learning_3
					has_trait = education_learning_4
					}
				}
			}
			shihao_5th_effect = yes
		}
		else_if = {
			limit = {
				OR = {
					has_trait = temperate
					has_trait = content
				}
				OR = {
					martial >= 10
					diplomacy >= 10
					intrigue >= 10
					stewardship >= 10
					learning >= 10
				}
			}
			shihao_6th_effect = yes
		}
		else_if = {
			limit = {
				age < 18
			}
			shihao_7th_effect = yes
		}
		else_if = {
			limit = {
				is_imprisoned = yes
			}
			give_nickname = "shihao_dao"
		}
		else = {
			shihao_8th_effect = yes
		}
	}
}
