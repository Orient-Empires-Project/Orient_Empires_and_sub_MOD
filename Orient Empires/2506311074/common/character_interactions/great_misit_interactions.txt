great_misit_kill_liege_interaction = {
	category = interaction_category_diarch
	#common_interaction = yes
	#diarch_interaction = yes
	
	interface_priority = 80
	greeting = positive
	icon = icon_entrench_regency
	desc = great_misit_kill_liege_interaction_desc
	ai_frequency = 480
	is_shown = {
		scope:actor = {
			is_vassal_of = scope:recipient
			OR = {
				has_trait = great_misit
				AND = {
					exists = liege
					is_diarch_of_target = liege
					liege = {
						has_diarchy_active_parameter = diarchy_is_primeministership
					}
				}
			}
			is_independent_ruler = no
			is_ai = no
			#faith = { religion_tag = hm_religion }
			#is_liege_or_above_of = scope:recipient
		}
		#scope:recipient = { 
			#has_relation_great_misit = scope:actor
		#}
	}
	is_valid_showing_failures_only = {
		scope:actor = {
			OR = {
				custom_description = {
					text = has_misit_power_100_tooltip
					has_variable = misit_power
					var:misit_power >= 100
				}
				liege ?= { has_diarchy_active_parameter = great_misit_kill_liege_interaction }
			}
			is_imprisoned = no
			is_incapable = no
		}
	}
		
	on_accept = {
		scope:actor = { 
			custom_tooltip = subtract_misit_power_100_tooltip
			change_variable = {
				name = misit_power
				subtract = 100
			}
			trigger_event = {
				id = great_misit.0001
				days = 3
			}
		}
	}
	auto_accept = yes
	ai_will_do = {
		base = 0
	}
}

great_misit_request_vassal_interaction = {
	category = interaction_category_diarch
	#common_interaction = yes
	#diarch_interaction = yes
	
	interface_priority = 60
	greeting = positive
	icon = icon_entrench_regency	
	desc = great_misit_request_vassal_interaction_desc
	needs_recipient_to_open = yes
	ai_frequency = 480
	populate_actor_list = {
		scope:recipient = {
			every_vassal = {
				limit = {
					is_ruler = yes
					highest_held_title_tier <= 4
					highest_held_title_tier >= 2
				}
				add_to_list = characters
			}
		}
	}
	is_shown = {
		scope:actor = {
			is_vassal_of = scope:recipient
			OR = {
				has_trait = great_misit
				AND = {
					exists = liege
					is_diarch_of_target = liege
					liege = {
						has_diarchy_active_parameter = diarchy_is_primeministership
					}
				}
			}
			is_independent_ruler = no
			is_ai = no
			#faith = { religion_tag = hm_religion }
			#is_liege_or_above_of = scope:recipient
		}
		#scope:recipient = { 
			#has_relation_great_misit = scope:actor
		#}
	}
	has_valid_target_showing_failures_only = {
		scope:secondary_actor = {
			primary_title.tier <= scope:actor.primary_title.tier
		}
	}
	is_valid_showing_failures_only = {
		scope:actor = {
			OR = {
				custom_description = {
					text = has_misit_power_200_tooltip 
					has_variable = misit_power
					var:misit_power >= 200
				}
				liege ?= { has_diarchy_active_parameter = great_misit_request_vassal_interaction }
			}
			is_imprisoned = no
			is_incapable = no
		}
	}
		
	on_accept = {
		scope:actor = {
			if = {
				limit = {
					scope:secondary_actor.primary_title.tier = tier_kingdom
				}
				scope:actor = { 
					custom_tooltip = subtract_misit_power_100_tooltip
					change_variable = {
						name = misit_power
						subtract = 40
					}
				}
			}
			if = {
				limit = {
					scope:secondary_actor.primary_title.tier >= tier_duchy
				}
				scope:actor = { 
					custom_tooltip = subtract_misit_power_60_tooltip
					change_variable = {
						name = misit_power
						subtract = 60
					}
				}
			}
			if = {
				limit = {
					scope:secondary_actor.primary_title.tier = tier_county
				}
				scope:actor = { 
					custom_tooltip = subtract_misit_power_25_tooltip
					change_variable = {
						name = misit_power
						subtract = 30
					}
				}
			}
			trigger_event = {
				id = great_misit.0002
				days = 3
			}
		}
	}
	auto_accept = yes
	ai_will_do = {
		base = 0
	}
}
