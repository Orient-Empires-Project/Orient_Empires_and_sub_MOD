admin_change_government_effect = {
	# Make sure you only use this effect on characters you are sure should change government! Since we don't do any extended triggers here.
	# For a consistent behaviour, this effect is based on "convert_whole_realm_to_feudalism_effect"
	
	# First, let's see if the character only has their noble family title - then we don't bother changing their government
	# We only destroy the family title to make sure it doesn't linger
	if = {
		limit = {
			is_landed = no
			any_held_title = {
				is_noble_family_title = yes #仅限原版世族
			}
			NOT = {
				any_held_title = {
					is_noble_family_title = no
				}
			}
		}
		every_held_title = {
			limit = {
				is_noble_family_title = yes
			}
			save_temporary_scope_as = family_title
			prev = { destroy_title = scope:family_title }  #销毁家族头衔？
		}
	}

	########
	else_if = {
		limit = {
			is_independent_ruler = no
			OR = {
				liege = { government_has_flag = government_is_celestial }
				liege = { government_has_flag = government_is_ulus }
			}
		}
		change_to_celes_government_effect = yes
	}
	else_if = {
		limit = {
			is_independent_ruler = no
			NOT = { government_has_flag = government_is_celestial }
			OR = {
				liege = { government_has_flag = government_is_legion }
			}
		}
		change_government = legion_government
	}
	else_if = {
		limit = {
			is_independent_ruler = no
			NOT = { government_has_flag = government_is_celestial }
			OR = {
				liege = { government_has_flag = government_is_ulus }
			}
		}
		change_government = ulus_government
	}
	
	# If not, let's change the government type
	# If liege has clan - Become clan!
	else_if = {
		limit = {
			is_independent_ruler = no
			NOT = { government_has_flag = government_is_celestial }
			OR = {
				liege = { government_has_flag = government_is_clan }
				AND = {
					liege = { government_has_flag = government_is_tribal }
					religion = religion:islam_religion
				}
			}
		}
		change_government = clan_government
	}
	# If liege has feudal - Become feudal!
	else_if = {
		limit = {
			is_independent_ruler = no
			NOT = { government_has_flag = government_is_celestial }
			OR = {
				liege = { government_has_flag = government_is_feudal }
				AND = {
					liege = { government_has_flag = government_is_tribal }
					NOT = { religion = religion:islam_religion }
				}
			}
		}
		change_government = feudal_government
	}
	#华夏传承独立转律令  （独立公爵级行政化？
	else_if = {
		limit = {
			is_independent_ruler = yes
			is_legion_government_trigger = yes
		}
		change_government = legion_government
	}
	else_if = {
		limit = {
			is_independent_ruler = yes
			culture = { has_cultural_pillar = heritage_chinese }
		}
		change_to_celes_government_effect = yes
	}
	# If you're an independent Muslim ruler - Become clan!
	else_if = {
		limit = {
			is_independent_ruler = yes
			faith = { religion_tag = islam_religion }
		}
		change_government = clan_government
	}
	# If you're an independent ruler with anything else - Become feudal!
	else = { change_government = feudal_government }
	
	# Finally, let's double-check and see if there are any remaining family titles we should destroy
	if = {
		limit = {
			NOT = { government_has_flag = government_use_estate }
			any_held_title = {
				is_oe_noble_family_title = yes
			}
		}
		every_held_title = {
			limit = {
				is_oe_noble_family_title = yes
			}
			save_temporary_scope_as = family_title
			prev = { destroy_title = scope:family_title }
		}
	}
}


change_to_administrative_effect = {
	save_scope_as = governor
	if = {
		limit = {
			liege = { government_has_flag = government_is_celestial }
			NOT = { government_has_flag = government_is_celestial }
		}
		change_government_legion_government_effect = yes
		save_scope_as = new_admin
	}
	else_if = {
		limit = {
			NOT = { government_has_flag = government_is_celestial }
			NOT = { government_has_flag = government_is_administrative }
		}
		change_government = administrative_government
		save_scope_as = new_admin
	}
	if = {
		limit = {
			primary_title.tier >= tier_duchy
			liege = {
				is_independent_ruler = yes
				government_has_flag = government_use_estate
			}
			house.house_head = {
				NOT = {
					any_held_title = { is_oe_noble_family_title = yes }
				}
			}
		}
		create_noble_family_effect = yes
	}
	else_if = {
		limit = {
			primary_title.tier >= tier_duchy
			liege = {
				is_independent_ruler = yes
				allows_imperial_bureaucracy = yes
				government_allows = administrative
			}
			house.house_head = {
				NOT = {
					any_held_title = { is_noble_family_title = yes }
				}
			}
		}
		create_noble_family_effect = yes
	}

	hidden_effect = {
		if = {
			limit = {
				is_governor = yes
				government_has_flag = government_is_celestial
				NOT = { has_trait = lifestyle_bureaucrat }
			}
			add_trait = lifestyle_bureaucrat
		}
		else_if = {
			limit = {
				is_governor = yes
				NOT = { has_trait = governor }
			}
			add_trait = governor
		}
	}
	if = {
		limit = { exists = scope:new_admin }
		domicile ?= {
			set_up_domicile_estate_effect = yes
		}
	}
}

	if = {
		limit = {
			NOT = { has_dlc_feature = roads_to_power }
			game_start_date = $DATE$
		}		
		create_title_and_vassal_change = {
			type = appointment
			save_scope_as = change
		}
		title:$TITLE$ = {
			change_title_holder = {
				holder = holder
				change = scope:change
			}
		}
		resolve_title_and_vassal_change = change
		holder = { destroy_title = prev }
	}
}

#完整式
oe_create_noble_family_effect = {
	debug_log = create_noble_family_title
	save_scope_as = new_noble_family_holder
	hidden_effect = { # NF can only be held by house heads
	if = {
		limit = {
			NOT = {
				any_held_title = { is_noble_family_title = yes }
			}
		}
		house ?= {
			if = {
				limit = { house_head != scope:new_noble_family_holder }
				set_house_head = scope:new_noble_family_holder
			}
		}

		#直接用原版贵族头衔封装效果，用于创建庄园
		give_noble_family_title = {
			name = noble_family_name
			article = DEFAULT_TITLE_NAME_ARTICLE
			save_scope_as = new_title
		}

		if = {
			limit = {
				government_has_flag = government_use_estate	
				NOR = {
					any_held_title = { is_oe_aristocrat_title = yes }
					house = {
						any_house_member = {
							any_held_title = { is_oe_aristocrat_title = yes }
						}
					}
				}
			}
			create_dynamic_title = {
				tier = duchy
				name = oe_noble_family_name
			}
			create_title_and_vassal_change = {
				type = created
				save_scope_as = change
				add_claim_on_loss = no
			}
			scope:new_title = {
				change_title_holder = {
					holder = scope:new_noble_family_holder
					change = scope:change
				}
				set_coa = scope:new_noble_family_holder.house
				set_capital_county = scope:new_noble_family_holder.location.county
				set_definitive_form = yes
				set_landless_title = yes
				set_no_automatic_claims = yes
			}
			resolve_title_and_vassal_change = scope:change
		}

		scope:new_title = {
			set_variable = {
				name = oe_aristocrat_title
				value = yes
			}
			holder = {
				save_scope_as = new_family
				save_scope_as = noble_family_head
			}
			add_title_law = noble_family_succession_law #添加继承法
			add_title_law = male_preference_law
		}
	}


	custom_description = {
		text = create_noble_family_tt
	}

	#Notifications
	top_liege ?= {
		every_vassal = {
			limit = {
				is_ai = no
				NOT = { this = scope:new_noble_family_holder }
			}
			trigger_event = {
				id = ep3_emperor_yearly.2410
				days = 1
			}
		}
	}
	}
	debug_log_scopes = yes
}
