JP_GetUIName = {
	type = character
	text = {
		localization_key = Loc_JP_GetUIName_default
		trigger = {
			is_local_player = yes
		}
	}
	text = {
		localization_key = Loc_JP_GetUIName
		trigger = {
			OR = {
				primary_title = title:k_yamato
				primary_title = title:d_gosho
				primary_title = title:d_sekke
			}
			has_any_nickname = no
		}
	}
	text = {
		localization_key = Loc_JP_GetUIName_nicknamed
		trigger = {
			OR = {
				primary_title = title:k_yamato
				primary_title = title:d_gosho
				primary_title = title:d_sekke
			}
			has_any_nickname = yes
		}
	}
	text = {
		localization_key = Loc_JP_GetUIName_firstname
		trigger = {
			OR = {
				AND = {
					is_female = yes
					is_landed = no
				}
				is_lowborn = yes
			}
			has_any_nickname = no
		}
	}
	text = {
		localization_key = Loc_JP_GetUIName_firstname_nicknamed
		trigger = {
			OR = {
				AND = {
					is_female = yes
					is_landed = no
				}
				is_lowborn = yes
			}
			has_any_nickname = yes
		}
	}
	text = {
		localization_key = Loc_JP_GetUIName_default
		fallback = yes
	}
}
