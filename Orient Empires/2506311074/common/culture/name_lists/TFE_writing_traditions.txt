tradition_script_arabic = {
	category = writing
	layers = {
		0 = scroll
		1 = mena
		3 = 2.dds
		4 = script_arabic.png
	}

	is_shown = {
		has_dlc_feature = royal_court
		OR = {
			scope:character = {
				any_realm_county = {
					culture = {
						has_cultural_tradition = tradition_script_arabic
					}
				}
			}
			scope:character = {
				liege.culture = {
					has_cultural_tradition = tradition_script_arabic
				}
			}
			this = culture:bedouin
			any_parent_culture_or_above = {
				this = culture:bedouin
			}
		}
		always = no
	}

	can_pick = {
		has_dlc_feature = royal_court
		no_other_script_trigger = yes
		#has_innovation = innovation_writing
		OR = {
			scope:character = {
				any_realm_county = {
					culture = {
						has_cultural_tradition = tradition_script_arabic
					}
				}
			}
			scope:character = {
				liege.culture = {
					has_cultural_tradition = tradition_script_arabic
				}
			}
			this = culture:bedouin
			any_parent_culture_or_above = {
				this = culture:bedouin
			}
		}
		always = no
	}

	can_pick_for_hybridization = {
		no_other_script_trigger = yes
	}

	character_modifier = {
		diplomacy = 1
		learning = 1
		diplomatic_range_mult = 0.5
		development_growth_factor = 0.1
		cultural_head_fascination_mult = 0.05
	}

	parameters = {
		grandeur_from_script = yes
	}

	cost = {
		prestige = {
			add = {
				value = 1000
				desc = BASE
				format = "BASE_VALUE_FORMAT"
			}
			if = {
				limit = {
					this = culture:bedouin
				}
				multiply = 0.5
			}
		}
		gold = {
			add = {
				value = 250
				desc = BASE
				format = "BASE_VALUE_FORMAT"
			}
			if = {
				limit = {
					this = culture:bedouin
				}
				multiply = 0.5
			}
		}
	}

	ai_will_do = {
		value = 1000
		if = {
			limit = {
				scope:character.liege.culture = {
					has_cultural_tradition = tradition_script_arabic
				}
			}
			multiply = 5
		}
		if = {
			limit = {
				this = culture:bedouin
			}
			multiply = 5
		}
	}
}
