language_coptic = {
	type = language
	is_shown = {
		language_is_shown_trigger = {
			LANGUAGE = language_coptic
		}
	}
	ai_will_do = {
		value = 10
		if = {
			limit = { has_cultural_pillar = language_coptic }
			multiply = 10
		}
		if = {
			limit = {
				scope:character = {
					faith = faith:coptic
				}
			}
			multiply = 2
		}
	}
	
	color = ancient_egyptian
}
