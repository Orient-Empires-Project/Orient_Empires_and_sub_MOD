shogunate_should_be_no_hair_trigger = {

	exists = this

	OR = {
		faith = faith:shinto
		faith = faith:shin
		faith = faith:tiantai
		faith = faith:kegon
		faith = faith:jingtu
		faith = faith:azhaliism
	}

	OR = {
		has_trait = devoted
		has_council_position = councillor_court_chaplain
		#shogunate_is_buddhist_trigger = yes
	}

}

portrait_japanese_wear_armor_trigger = {

	exists = this

	OR = {
		AND = {
			is_ruler = yes
			has_government = mercenary_government
		}
		is_commanding_army = yes
		exists = knight_army
		AND = {
			OR = {
				is_at_war = yes
				liege = {
					is_at_war = yes
				}
			}		
			OR = {
				AND = {
					is_knight = yes
					is_councillor = no
					OR = {
						is_landed = no
						liege = {
							is_at_war = yes
						}
					}
				}
				AND = {
					has_council_position = councillor_marshal
					OR = {
						is_landed = no
						primary_title.tier < tier_kingdom
					}
				}
			}
		}
		has_character_flag = is_currently_varangian
		has_character_flag = single_combat_duel_armor
	}

}
