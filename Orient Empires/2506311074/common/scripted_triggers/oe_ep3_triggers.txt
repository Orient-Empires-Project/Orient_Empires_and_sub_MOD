allows_imperial_bureaucracy = {
	OR = {
		government_allows = administrative
		government_has_flag = government_use_bureaucracy
	}
}
allows_administrative_NOT_celestial = {
	government_allows = administrative
	NOT = { government_has_flag = government_use_bureaucracy }
}
