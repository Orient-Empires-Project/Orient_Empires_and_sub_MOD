am_neishijian_list = {
	scope = character

	effect = {
		if = {
			limit = {
				root = {
					has_government = imperial_government
				}				
			}
			root = {
				## Clear variable always before
				clear_variable_list = neishijian_list
				every_courtier = {
					limit = {
						has_council_position = councillor_neishijian
					}
					root = {
						add_to_variable_list = {
							name = neishijian_list
							target = prev
						}
					}					
				}
				every_vassal = {
					limit = {
						has_council_position = councillor_neishijian
					}
					root = {
						add_to_variable_list = {
							name = neishijian_list
							target = prev
						}
					}					
				}
			}
		}
		if = {
			limit = {
				root = {
					NOT = {
						has_government = imperial_government
					}	
					any_liege_or_above = {
						has_government = imperial_government
					}				
				}
			}
			root = { clear_variable_list = neishijian_list }
			root.liege = {
				every_courtier = {
					limit = {
						has_council_position = councillor_neishijian
					}
					root = {
						add_to_variable_list = {
							name = neishijian_list
							target = prev
						}
					}					
				}
				every_vassal = {
					limit = {
						has_council_position = councillor_neishijian
					}
					root = {
						add_to_variable_list = {
							name = neishijian_list
							target = prev
						}
					}					
				}
			}
		}
	}
}

show_china_offices_button = {
	scope = character
	is_shown = {
		OR = {
			AND = {
				capital_province = { geographical_region = world_asia }
				highest_held_title_tier = tier_empire
			}
			any_liege_or_above = {
				capital_province = { geographical_region = world_asia }
				highest_held_title_tier = tier_empire
			}					
		}
	}
